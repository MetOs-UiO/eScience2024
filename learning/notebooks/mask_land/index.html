def convert_ds_lon_to_180(ds):
    ds['lon'] = (ds['lon']+ 180) % 360 - 180
    ds =ds.sortby('lon')
    return ds

ds_360 = ds.copy()
ds = convert_ds_lon_to_180(ds)
